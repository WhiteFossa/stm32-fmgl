83  FMGL_API_ColorStruct activeColor, FMGL_API_ColorStruct inactiveColor, FMGL_API_XBMTransparencyMode transparency)
84 {

86  if (x > context->MaxX || y > context->MaxY)

88  return;
89  }

95  colors[FMGL_PRIV_XBM_ACTIVE_COLOR_INDEX] = activeColor;
96  colors[FMGL_PRIV_XBM_INACTIVE_COLOR_INDEX] = inactiveColor;
97

99  {
100  if
101  (
102  ((FMGL_PRIV_XBM_ACTIVE_COLOR_INDEX == colorIndex) && (FMGL_XBMTransparencyModeTransparentActive == transparency))
103  ||
104  ((FMGL_PRIV_XBM_INACTIVE_COLOR_INDEX == colorIndex) && (FMGL_XBMTransparencyModeTransparentInactive == transparency))
105  )
106  {
107  /* Skipping transparent color */
108  continue;
109  }
110 
111  FMGL_API_SetActiveColor(context, colors[colorIndex]);
112 
113  for (uint32_t dy = 0; dy < scaledHeight; dy++)
114  {
115  uint16_t sy = dy / scaleY;
116 
117  for (uint32_t dx = 0; dx < scaledWidth; dx++)
118  {
119  uint16_t sx = dx / scaleX;
120 
121  bool isActive = FMGL_Priv_IsActiveXBMPixel(image, sx, sy);
122 
123  if
124  (
125  ((FMGL_PRIV_XBM_ACTIVE_COLOR_INDEX == colorIndex) && isActive)
126  ||
127  ((FMGL_PRIV_XBM_INACTIVE_COLOR_INDEX == colorIndex) && !isActive)
128  )
129  {
130  /* Active pixel */
131  FMGL_API_DrawPixel(context, dx + x, dy + y);
132  }
133  }

304 void FMGL_API_RenderOneLineDumb(FMGL_API_DriverContext* context, FMGL_API_FontSettings* fontSettings, uint16_t x, uint16_t y, uint16_t* width,
305  bool isDryRun, char* string)
306 {
307  uint16_t scaledCharactersSpacing = fontSettings->CharactersSpacing * fontSettings->Scale;
308  uint16_t currentX = x;
309 
310  char* currentChar = string;
311 
312  while (true)
313  {
314  if ('\0' == *currentChar)
315  {
316  /* End of line */
317  return;
318  }
319  else if ('\n' == *currentChar)
320  {
321  /* Newline is not allowed */
322  L2HAL_Error(L2HAL_ERROR_WRONG_ARGUMENT);
323  }
324 
325  /* Could we draw current pixel? */
326  if (currentX > context->MaxX)
327  {
328  /* No */
329  return;
330  }
331 
332  /* Drawing */
333  if (!isDryRun)
334  {
335  FMGL_Priv_RenderCharacter(context, fontSettings, currentX, y, *currentChar);
336  }
337 
338  currentX += fontSettings->Font->GetCharacterWidth(*currentChar);
339  *width = currentX - x - 1;
340 
341  /* Adding intercharacter spacing */
342  currentX += scaledCharactersSpacing;
343 
344  currentChar ++;
345  }
